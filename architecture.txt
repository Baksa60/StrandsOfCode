StrandsOfCode/
│
├─ main.py                     # Точка входа. Запуск приложения и инициализация PyQt6
├─ version.py                  # Версия и метаданные продукта
│
├─ ui/
│   ├─ main_window.py          # Главное окно приложения (Drag & Drop, выбор файлов, настройки)
│   ├─ dialogs.py              # Дополнительные диалоги (выбор файлов, сообщений, прогресса)
│   └─ styles.py              # Стили тем (темная/светлая тема и визуальные эффекты)
│
├─ controllers/
│   └─ convert_controller.py   # Контроллер: связывает UI и бизнес-логику
│                               # принимает параметры из интерфейса и запускает обработку
│
├─ services/
│   ├─ file_collector.py       # Сервис поиска файлов: файл / несколько / папка / рекурсивно
│   ├─ converter_service.py    # Базовый сервис конвертации: читает исходники и создаёт текст
│   ├─ html_converter_service.py    # Конвертация в HTML с CSS стилями и подсветкой синтаксиса
│   ├─ markdown_converter_service.py # Конвертация в Markdown формат
│   ├─ json_converter_service.py     # Конвертация в JSON формат
│   ├─ text_to_text_converter.py     # Универсальный конвертер текста
│   ├─ base_reverse_converter.py     # Базовый класс для обратной конвертации
│   ├─ code_to_code_converter.py     # Конвертер между языками программирования
│   ├─ txt_to_code_converter.py      # Восстановление кода из TXT
│   ├─ html_to_code_converter.py     # Восстановление кода из HTML
│   ├─ json_to_code_converter.py     # Восстановление кода из JSON
│   └─ markdown_to_code_converter.py # Восстановление кода из Markdown
│
├─ models/
│   ├─ conversion_options.py   # Модель параметров операции (что выбрано и как конвертировать)
│   └─ conversion_result.py    # Модель результата конвертации (статистика, файлы, метаданные)
│
├─ utils/
│   ├─ file_utils.py           # Низкоуровневые функции работы с файлами (чтение, запись, фильтрация)
│   └─ cancellation.py         # Утилиты отмены операций
│
├─ config.py                   # Глобальные настройки программы (расширения, маркеры проектов, фильтры)
├─ requirements.txt            # Зависимости проекта (PyQt6, chardet)
├─ ROADMAP.md                  # План развития проекта
├─ CHANGELOG.md                # История изменений
├─ README.md                   # Документация проекта
└── LICENSE                    # Лицензия MIT
